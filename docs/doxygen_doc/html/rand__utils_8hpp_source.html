152  /*
153  * seed_seq_fe implements a fixed-entropy seed sequence; it conforms to all
154  * the requirements of a Seed Sequence concept.
155  *
156  * seed_seq_fe<N> implements a seed sequence which seeds based on a store of
157  * N * 32 bits of entropy. Typically, it would be initialized with N or more
158  * integers.
159  *
160  * seed_seq_fe128 and seed_seq_fe256 are provided as convenience typedefs for
161  * 128- and 256-bit entropy stores respectively. These variants outperform
162  * std::seed_seq, while being better mixing the bits it is provided as entropy.
163  * In almost all common use cases, they serve as better drop-in replacements
164  * for seed_seq.
165  *
166  * Technical details
167  *
168  * Assuming it constructed with M seed integers as input, it exhibits the
169  * following properties
170  *
171  * * Diffusion/Avalanche: A single-bit change in any of the M inputs has a
172  * 50% chance of flipping every bit in the bitstream produced by generate.
173  * Initializing the N-word entropy store with M words requires O(N * M)
174  * time precisely because of the avalanche requirements. Once constructed,
175  * calls to generate are linear in the number of words generated.
176  *
177  * * Bias freedom/Bijection: If M == N, the state of the entropy store is a
178  * bijection from the M inputs (i.e., no states occur twice, none are
179  * omitted). If M > N the number of times each state can occur is the same
180  * (each state occurs 2**(32*(M-N)) times, where ** is the power function).
181  * If M < N, some states cannot occur (bias) but no state occurs more
182  * than once (it's impossible to avoid bias if M < N; ideally N should not
183  * be chosen so that it is more than M).
184  *
185  * Likewise, the generate function has similar properties (with the entropy
186  * store as the input data). If more outputs are requested than there is
187  * entropy, some outputs cannot occur. For example, the Mersenne Twister
188  * will request 624 outputs, to initialize it's 19937-bit state, which is
189  * much larger than a 128-bit or 256-bit entropy pool. But in practice,
190  * limiting the Mersenne Twister to 2**128 possible initializations gives
191  * us enough initializations to give a unique initialization to trillions
192  * of computers for billions of years. If you really have 624 words of
193  * *real* high-quality entropy you want to use, you probably don't need
194  * an entropy mixer like this class at all. But if you *really* want to,
195  * nothing is stopping you from creating a randutils::seed_seq_fe<624>.
196  *
197  * * As a consequence of the above properties, if all parts of the provided
198  * seed data are kept constant except one, and the remaining part is varied
199  * through K different states, K different output sequences will be produced.
200  *
201  * * Also, because the amount of entropy stored is fixed, this class never
202  * performs dynamic allocation and is free of the possibility of generating
203  * an exception.
204  *
205  * Ideas used to implement this code include hashing, a simple PCG generator
206  * based on an MCG base with an XorShift output function and permutation
207  * functions on tuples.
208  *
209  * More detail at
210  * http://www.pcg-random.org/posts/developing-a-seed_seq-alternative.html
211  */

232  static IntRep fast_exp(IntRep x, IntRep power)

236  while (power != IntRep(0)) {
237  IntRep thismult = power & IntRep(1) ? multiplier : IntRep(1);

239  power >>= 1;

295  template <size_t count, typename IntRep, size_t r>
296  template <typename InputIter>
297  void seed_seq_fe<count, IntRep, r>::mix_entropy(InputIter begin, InputIter end)
298  {
299  auto hash_const = INIT_A;
300  auto hash = [&](IntRep value) {
301  value ^= hash_const;
302  hash_const *= MULT_A;
303  value *= hash_const;
304  value ^= value >> XSHIFT;
305  return value;
306  };
307  auto mix = [](IntRep x, IntRep y) {
308  IntRep result = MIX_MULT_L*x - MIX_MULT_R*y;
309  result ^= result >> XSHIFT;
310  return result;
311  };
312 
313  InputIter current = begin;
314  for (auto& elem : mixer_) {
315  if (current != end)
316  elem = hash(*current++);
317  else
318  elem = hash(0U);
319  }
320  for (auto& src : mixer_)
321  for (auto& dest : mixer_)
322  if (&src != &dest)
323  dest = mix(dest, hash(src));
324  for (; current != end; ++current)
325  for (auto& dest : mixer_)
326  dest = mix(dest, hash(*current));
327  }

329  template <size_t count, typename IntRep, size_t mix_rounds>
330  template <typename OutputIterator>
331  void seed_seq_fe<count, IntRep, mix_rounds>::param(OutputIterator dest) const
332  {
333  const IntRep INV_A = fast_exp(MULT_A, IntRep(-1));
334  const IntRep MIX_INV_L = fast_exp(MIX_MULT_L, IntRep(-1));
335 
336  auto mixer_copy = mixer_;
337  for (size_t round = 0; round < mix_rounds; ++round) {
338  // Advance to the final value. We'll backtrack from that.
339  auto hash_const = INIT_A * fast_exp(MULT_A, IntRep(count * count));
340 
341  for (auto src = mixer_copy.rbegin(); src != mixer_copy.rend(); ++src)
342  for (auto dest = mixer_copy.rbegin(); dest != mixer_copy.rend();
343  ++dest)
344  if (src != dest) {
345  IntRep revhashed = *src;
346  auto mult_const = hash_const;
347  hash_const *= INV_A;
348  revhashed ^= hash_const;
349  revhashed *= mult_const;
350  revhashed ^= revhashed >> XSHIFT;
351  IntRep unmixed = *dest;
352  unmixed ^= unmixed >> XSHIFT;
353  unmixed += MIX_MULT_R * revhashed;
354  unmixed *= MIX_INV_L;
355  *dest = unmixed;
356  }
357  for (auto i = mixer_copy.rbegin(); i != mixer_copy.rend(); ++i) {
358  IntRep unhashed = *i;
359  unhashed ^= unhashed >> XSHIFT;
360  unhashed *= fast_exp(hash_const, IntRep(-1));
361  hash_const *= INV_A;
362  unhashed ^= hash_const;
363  *i = unhashed;
364  }
365  }
366  std::copy(mixer_copy.begin(), mixer_copy.end(), dest);
367  }

370  template <typename RandomAccessIterator>
371  void seed_seq_fe<count, IntRep, mix_rounds>::generate(
372  RandomAccessIterator dest_begin,
373  RandomAccessIterator dest_end) const
374  {
375  auto src_begin = mixer_.begin();
376  auto src_end = mixer_.end();
377  auto src = src_begin;
378  auto hash_const = INIT_B;
379  for (auto dest = dest_begin; dest != dest_end; ++dest) {
380  auto dataval = *src;
381  if (++src == src_end)
382  src = src_begin;
383  dataval ^= hash_const;

385  dataval *= hash_const;
386  dataval ^= dataval >> XSHIFT;
387  *dest = dataval;
388  }
