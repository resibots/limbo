132  struct Cmaes {
133  public:
134  template <typename F>
135  Eigen::VectorXd operator()(const F& f, const Eigen::VectorXd& init, double bounded) const
136  {
137  size_t dim = init.size();
138 
139  // wrap the function
140  libcmaes::FitFunc f_cmaes = [&](const double* x, const int n) {
141  Eigen::Map<const Eigen::VectorXd> m(x, n);
142  // remember that our optimizers maximize
143  return -eval(f, m);
144  };
145 
146  if (bounded)
147  return _opt_bounded(f_cmaes, dim, init);
148  else
149  return _opt_unbounded(f_cmaes, dim, init);
150  }
151 
152  private:
153  // F is a CMA-ES style function, not our function
154  template <typename F>
155  Eigen::VectorXd _opt_unbounded(F& f_cmaes, int dim, const Eigen::VectorXd& init) const
156  {
157  using namespace libcmaes;
158  // initial step-size, i.e. estimated initial parameter error.
159  double sigma = 0.5;
160  std::vector<double> x0(init.data(), init.data() + init.size());
161 
162  CMAParameters<> cmaparams(x0, sigma);
163  _set_common_params(cmaparams, dim);
164 
165  // the optimization itself
166  CMASolutions cmasols = cmaes<>(f_cmaes, cmaparams);
167  return cmasols.get_best_seen_candidate().get_x_dvec();
168  }

170  // F is a CMA-ES style function, not our function
171  template <typename F>
172  Eigen::VectorXd _opt_bounded(F& f_cmaes, int dim, const Eigen::VectorXd& init) const
173  {
174  using namespace libcmaes;
175  // create the parameter object
176  // boundary_transformation
177  double lbounds[dim], ubounds[dim]; // arrays for lower and upper parameter bounds, respectively
178  for (int i = 0; i < dim; i++) {
179  lbounds[i] = Params::opt_cmaes::lbound();
180  ubounds[i] = Params::opt_cmaes::ubound();
181  }
182  GenoPheno<pwqBoundStrategy> gp(lbounds, ubounds, dim);
183  // initial step-size, i.e. estimated initial parameter error.
184  double sigma = 0.5 * std::abs(Params::opt_cmaes::ubound() - Params::opt_cmaes::lbound());
185  std::vector<double> x0(init.data(), init.data() + init.size());
186  // -1 for automatically decided lambda, 0 is for random seeding of the internal generator.
187  CMAParameters<GenoPheno<pwqBoundStrategy>> cmaparams(dim, &x0.front(), sigma, -1, 0, gp);
188  _set_common_params(cmaparams, dim);
189 
190  // the optimization itself
191  CMASolutions cmasols = cmaes<GenoPheno<pwqBoundStrategy>>(f_cmaes, cmaparams);
192  //cmasols.print(std::cout, 1, gp);
193  //to_f_representation
194  return gp.pheno(cmasols.get_best_seen_candidate().get_x_dvec());
195  }

197  template <typename P>
198  void _set_common_params(P& cmaparams, int dim) const
199  {
200  using namespace libcmaes;
201 
202  // set multi-threading to true
203  cmaparams.set_mt_feval(true);
204  // aCMAES should be the best choice
205  // [see: https://github.com/beniz/libcmaes/wiki/Practical-hints ]
206  // but we want the restart -> aIPOP_CMAES
207  cmaparams.set_algo(Params::opt_cmaes::variant());
208  cmaparams.set_restarts(Params::opt_cmaes::restarts());
209  cmaparams.set_elitism(Params::opt_cmaes::elitism());
210 
211  // if no max fun evals provided, we compute a recommended value
212  size_t max_evals = Params::opt_cmaes::max_fun_evals() < 0
213  ? (900.0 * (dim + 3.0) * (dim + 3.0))
214  : Params::opt_cmaes::max_fun_evals();
215  cmaparams.set_max_fevals(max_evals);
216  // max iteration is here only for security
217  cmaparams.set_max_iter(100000);
218 
219  if (Params::opt_cmaes::fun_tolerance() == -1) {
220  // we do not know if what is the actual maximum / minimum of the function
221  // therefore we deactivate this stopping criterion
222  cmaparams.set_stopping_criteria(FTARGET, false);
223  }
224  else {
225  // the FTARGET criteria also allows us to enable ftolerance
226  cmaparams.set_stopping_criteria(FTARGET, true);
227  cmaparams.set_ftolerance(Params::opt_cmaes::fun_tolerance());
228  }
229 
230  // we allow to set the ftarget parameter
231  if (Params::opt_cmaes::fun_target() != -1) {
232  cmaparams.set_stopping_criteria(FTARGET, true);
233  cmaparams.set_ftarget(-Params::opt_cmaes::fun_target());
234  }
235 
236  // enable stopping criteria by several equalfunvals and maxfevals
237  cmaparams.set_stopping_criteria(EQUALFUNVALS, true);
238  cmaparams.set_stopping_criteria(MAXFEVALS, true);
239 
240  // enable additional criterias to stop
241  cmaparams.set_stopping_criteria(TOLX, true);
242  cmaparams.set_stopping_criteria(CONDITIONCOV, true);
243 
244  // enable or disable different parameters
245  cmaparams.set_initial_fvalue(Params::opt_cmaes::fun_compute_initial());
246  cmaparams.set_uh(Params::opt_cmaes::handle_uncertainty());
247  cmaparams.set_quiet(!Params::opt_cmaes::verbose());
248  }
249  };
